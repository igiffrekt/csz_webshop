  <action>
Create production Docker Compose configuration:

```yaml
version: '3.8'

services:
  postgres:
    image: postgres:16-alpine
    restart: always
    environment:
      POSTGRES_DB: ${POSTGRES_DB:-csz_webshop}
      POSTGRES_USER: ${POSTGRES_USER:-csz}
      POSTGRES_PASSWORD: ${POSTGRES_PASSWORD}
    volumes:
      - postgres_data:/var/lib/postgresql/data
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U ${POSTGRES_USER:-csz}"]
      interval: 10s
      timeout: 5s
      retries: 5

  cms:
    build:
      context: ./apps/cms
      dockerfile: Dockerfile
    restart: always
    environment:
      DATABASE_CLIENT: postgres
      DATABASE_URL: ${DATABASE_URL}
      NODE_ENV: production
    depends_on:
      postgres:
        condition: service_healthy
    ports:
      - "1337:1337"

  api:
    build:
      context: ./apps/api
      dockerfile: Dockerfile
    restart: always
    environment:
      NODE_ENV: production
      DATABASE_URL: ${DATABASE_URL}
      STRAPI_URL: ${STRAPI_URL}
    depends_on:
      - cms
    ports:
      - "3001:3001"

  web:
    build:
      context: ./apps/web
      dockerfile: Dockerfile
    restart: always
    environment:
      NODE_ENV: production
    depends_on:
      - api
      - cms
    ports:
      - "3000:3000"

volumes:
  postgres_data:

networks:
  default:
    name: csz_network
```

Create Dockerfile for web app:

```dockerfile
# apps/web/Dockerfile
FROM node:20-alpine AS base

# Install dependencies only when needed
FROM base AS deps
RUN apk add --no-cache libc6-compat
WORKDIR /app

COPY package.json pnpm-lock.yaml ./
RUN corepack enable pnpm && pnpm install --frozen-lockfile

# Build the app
FROM base AS builder
WORKDIR /app
COPY --from=deps /app/node_modules ./node_modules
COPY . .

ENV NEXT_TELEMETRY_DISABLED 1

RUN corepack enable pnpm && pnpm build

# Production image
FROM base AS runner
WORKDIR /app

ENV NODE_ENV production
ENV NEXT_TELEMETRY_DISABLED 1

RUN addgroup --system --gid 1001 nodejs
RUN adduser --system --uid 1001 nextjs

COPY --from=builder /app/public ./public
COPY --from=builder --chown=nextjs:nodejs /app/.next/standalone ./
COPY --from=builder --chown=nextjs:nodejs /app/.next/static ./.next/static

USER nextjs

EXPOSE 3000

ENV PORT 3000

CMD ["node", "server.js"]
```

## Deployment

### Prerequisites
- Docker and Docker Compose
- PostgreSQL database
- Stripe account (live keys)
- SMTP email service
- Domain with SSL

### Environment Setup
1. Copy `.env.production.example` to `.env.production`
2. Fill in all required values
3. Generate secure secrets for JWT and session

### Deploy with Docker
```bash
# Build images
docker-compose -f docker-compose.prod.yml build

# Start services
docker-compose -f docker-compose.prod.yml up -d

# Run migrations
docker-compose -f docker-compose.prod.yml exec cms pnpm strapi database:migrate
```

### Vercel Deployment (Web)
```bash
cd apps/web
vercel --prod
```

### Post-Deployment
1. Configure DNS records
2. Set up SSL certificates
3. Configure Stripe webhooks
4. Run data migration
5. Test all critical flows
